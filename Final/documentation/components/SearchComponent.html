import { Component } from '@angular/core';

@Component({
  selector: 'app-search',
  templateUrl: './search.component.html',
  styleUrls: ['./search.component.css']
})
export class SearchComponent {

  search = 'call'; // Default communication option
  searchBy = 'location'; // Default search option
  date: Date | undefined; // Date input

  showDateInput = false; // Flag to show/hide date input
  searchLocation: any;
  searchName: any;
  searchNumber: any;

  onCommunicationOptionChange(option: string) {
    this.search = option;
  }

  onSearchByChange(option: string) {
    this.searchBy = option;

    // Show date input only for "Date" option
    this.showDateInput = option === 'date';
  }

  onSubmit() {
    // Handle form submission
    console.log('Form submitted');
  }
}
