import { Component } from '@angular/core';
import { FormBuilder } from '@angular/forms';
import { MatDialog } from '@angular/material/dialog';
import { CustomersService } from '../customers.service';
import { MessagedialogComponent } from '../messagedialog/messagedialog.component';

@Component({
  selector: 'app-uploadaccount',
  templateUrl: './uploadaccount.component.html',
  styleUrls: ['./uploadaccount.component.css']
})
export class UploadaccountComponent {

  selectedFile: File | null = null;

  constructor(private customersService: CustomersService, public dialog: MatDialog, private formBuilder: FormBuilder) { }
  onFileSelected(event: any) {
    this.selectedFile = event.target.files[0];
  }
  uploadCustomersFile() {
    console.log("gvhfv");

    if (this.selectedFile) {
      const formData = new FormData();
      formData.append('file', this.selectedFile);
      this.customersService.uploadCustomersCSVFile(formData).subscribe(response => {
        console.log("uploadedddddddddddddddd")
        if (response.status === 'uploaded successfully') {
          this.openDialog('File Upload', 'CSV file uploaded and processed successfully.');
        } else {
          this.openDialog('File Upload', 'Error occurred while processing the CSV file.');
        }
      });
    } else {

      this.openDialog('File Upload', 'Please select a CSV file before uploading.');
    }
  }

  openDialog(title: string, message: string): void {
    const dialogRef = this.dialog.open(MessagedialogComponent, {
      width: '250px',
      data: { title, message }
    });

    dialogRef.afterClosed().subscribe(result => {
      console.log('The dialog was closed');
    });
  }

}
